# Open a ebpage
OPEN "https://example.com"


ON "#login-form" {
    FILL "#username" "test_user"
    FILL "#password" "password123"
    CLICK "#submit"
}

# Read value and store in variable
READ "#welcome-message" AS welcome_text


IF welcome_text HAS "Hello, test_user" {
    CLICK "#continue"
} ELSE {
    CLICK "#retry"
}

# Looping Keep checking if element exists
WHILE "#loading" EXISTS {
    WAIT 1s
}


TRY {
    CLICK "#confirm"
} CATCH {
    LOG "Failed to click confirm button."
}
